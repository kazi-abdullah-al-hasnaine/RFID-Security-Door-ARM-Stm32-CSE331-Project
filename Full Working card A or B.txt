#include <SPI.h>
#include <MFRC522.h>
#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

#define RST_PIN     PB0    // RC522 RST
#define SS_PIN      PA4    // RC522 NSS (SDA)
#define LED_PIN     PC13   // Onboard LED (active LOW)

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
#define OLED_RESET    -1

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);
MFRC522 mfrc522(SS_PIN, RST_PIN);   // Create MFRC522 instance

// Card UIDs (change to your own)
byte cardA[4] = {0x06, 0xCC, 0x87, 0x97};
byte cardB[4] = {0x03, 0x96, 0x41, 0xC5};

void setup() {
  pinMode(LED_PIN, OUTPUT);
  digitalWrite(LED_PIN, HIGH);  // LED off initially

  Serial.begin(9600);
  SPI.begin();
  mfrc522.PCD_Init(); // Init RC522
  delay(50);

  Wire.begin();
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    while (1); // OLED init failed
  }

  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(WHITE);
  display.setCursor(0, 0);
  display.println("RFID + OLED Ready");
  display.display();
  delay(1000);
}

void loop() {
  // Show "Waiting..." message
  display.clearDisplay();
  display.setCursor(0, 0);
  display.println("Waiting...");
  display.display();

  // Wait for a new card
  if (!mfrc522.PICC_IsNewCardPresent() || !mfrc522.PICC_ReadCardSerial()) {
    delay(200);
    return;
  }

  // Card detected
  Serial.print("Card UID: ");
  for (byte i = 0; i < mfrc522.uid.size; i++) {
    Serial.print(mfrc522.uid.uidByte[i] < 0x10 ? "0" : "");
    Serial.print(mfrc522.uid.uidByte[i], HEX);
    Serial.print(" ");
  }
  Serial.println();

  // Compare UID
  bool isCardA = compareUID(mfrc522.uid.uidByte, cardA);
  bool isCardB = compareUID(mfrc522.uid.uidByte, cardB);

  display.clearDisplay();
  display.setCursor(0, 0);

  if (isCardA) {
    display.println("Card A matched");
    Serial.println("Card A matched");
    blinkLED(150);
  } else if (isCardB) {
    display.println("Card B matched");
    Serial.println("Card B matched");
    blinkLED(150);
  } else {
    display.println("Invalid card");
    Serial.println("Invalid card");
    blinkLED(400);  // Longer blink for invalid
  }

  display.display();
  delay(1500);

  mfrc522.PICC_HaltA();
  mfrc522.PCD_StopCrypto1();
}

// Compare two 4-byte UIDs
bool compareUID(byte *uid, byte *validUID) {
  for (int i = 0; i < 4; i++) {
    if (uid[i] != validUID[i]) return false;
  }
  return true;
}

void blinkLED(int duration) {
  digitalWrite(LED_PIN, LOW);  // LED ON
  delay(duration);
  digitalWrite(LED_PIN, HIGH); // LED OFF
}
